CREATE TABLE `quintuplecache` (
  `ChampOne` int DEFAULT NULL,
  `ChampTwo` int DEFAULT NULL,
  `ChampThree` int DEFAULT NULL,
  `ChampFour` int DEFAULT NULL,
  `ChampFive` int DEFAULT NULL,
  `Patch` varchar(20) DEFAULT NULL,
  `Region` varchar(20) DEFAULT NULL,
  `Wins` bigint DEFAULT NULL,
  `Games` bigint DEFAULT NULL,
  `AverageDuration` float DEFAULT NULL COMMENT 'This column stores the duration of the match in seconds',
  UNIQUE KEY `identifying_index` (`ChampOne`,`ChampTwo`, `ChampThree`, `ChampFour`, `ChampFive`, `Region`,`Patch`),
  KEY `lookup` (`ChampOne`,`ChampTwo`, `ChampThree`, `ChampFour`, `ChampFive`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='This table caches full team results';
